script final :


#! /bin/sh

echo $# '-' $@

args=""

while (($#))
do
    while getopts ":hf:" opt 
    do
        case $opt in
            f) echo "f" $OPTIND '-' $OPTARG
                ;;
            h) echo "h" $OPTIND '-' $OPTARG
                ;;
            :) echo "option manquante" $OPTIND '-' $OPTARG
                ;;
            \?) echo "inconnu" $OPTIND '-' $OPTARG
                ;;
        esac
    done
    shift $((OPTIND-1))
    echo $# '-' $@
    if [ $# -gt 0 ]
    then
        args=$args" $1"
        shift
    fi
done

echo "args:"$args"-"
